<script>
    // Function to detect phishing URL
    function detectURL() {
        var url = document.getElementById("urlInput").value.trim();
        var resultElement = document.getElementById("result");

        // Clear previous results
        resultElement.innerHTML = "";

        // Validate URL
        if (url === "") {
            showError("Please enter a URL.");
            return;
        }

        // Check if the URL starts with "https" or "www"
        var isPhishing = inputIsPhishing(url);

        // Display result or error message
        if (isPhishing === null) {
            showResult(`You entered an incorrect URL. Please check again.`, "error");
        } else if (!isPhishing) {
            showResult(`The URL ${url} is identified as a legitimate site.`, "success");
        } else {
            showResult(`The URL ${url} is identified as a phishing site.`, "error");
        }
    }

    // Function to check if input is a phishing URL
    function inputIsPhishing(url) {
        // Check if the input is a valid URL
        try {
            new URL(url);
        } catch (error) {
            return null; // Not a valid URL
        }

        // Check if the URL starts with "https" or "www"
        if (url.startsWith("https://") || url.startsWith("www.")) {
            return false; // It's a legitimate website
        } else if (url.startsWith("http://")) {
            return true; // It's a phishing website
        } else {
            return true; // It's not a valid URL (consider it phishing)
        }
    }

    // Function to clear the form
    function clearForm() {
        document.getElementById("urlInput").value = ""; // Clear input field
        document.getElementById("result").innerHTML = ""; // Clear result message
        clearError(); // Clear any error message
    }

    function submitForm(event) {
        event.preventDefault(); // Prevent form submission
        var url = document.getElementById("urlInput").value.trim();


        // Send AJAX request to Flask backend
        $.ajax({
            type: "POST",
            url: "/save-url",
            contentType: "application/json",
            data: JSON.stringify({"url": url}),
            success: function (response) {
                console.log("URL saved to database:", response);
                // Optionally, display a success message to the user
                // Display success message
                var resultContainer = document.getElementById("result");
                resultContainer.innerHTML = '<p class="text-success">URL saved successfully!</p>';
            },
            error: function (errorz) {
                console.error("Error saving URL:", error);
                // Optionally, display an error message to the user
                var resultContainer = document.getElementById("result");
                resultContainer.innerHTML = '<p class="text-danger">Error saving URL. Please try again.</p>';
            }
        });
    }

    // Function to display error message
    function showError(message) {
        var errorElement = document.getElementById("urlError");
        errorElement.textContent = message;
        errorElement.style.display = "block";
    }

    // Function to clear error message
    function clearError() {
        var errorElement = document.getElementById("urlError");
        errorElement.textContent = "";
        errorElement.style.display = "none";
    }

    // Function to display result message
    function showResult(message, messageType) {
        var resultContainer = document.getElementById("result");
        var resultMessage = '<p class="';

        // Determine message type
        if (messageType === "error") {
            resultMessage += 'text-danger">';
        } else if (messageType === "success") {
            resultMessage += 'text-success">';
        } else {
            resultMessage += 'text-warning">';
        }

        resultMessage += message + '</p>';

        resultContainer.innerHTML = resultMessage;
    }
</script>
